🔧 مقارنة شاملة للفلاتر

فلاتر النصوص

الفلتر DTL Jinja2 الوصف مثال الناتج
تحويل لحروف كبيرة {{ "hello"|upper }} {{ "hello"|upper }} تحويل كل الحروف لكبيرة HELLO
تحويل لحروف صغيرة {{ "HELLO"|lower }} {{ "HELLO"|lower }} تحويل كل الحروف لصغيرة hello
أول حرف كبير {{ "hello"|capfirst }} {{ "hello"|capfirst }} تحويل أول حرف لكبير Hello
اقتطاع كلمات {{ text|truncatewords:5 }} {{ text|truncatechars:5 }} اقتطاع النص بعد عدد الكلمات هذا النص طويل... (إذا كان 5 كلمات)
إزالة مسافات {{ " hello "|cut:" " }} { " hello "| trim } إزالة المسافات الزائدة hello

فلاتر الأرقام

الفلتر DTL Jinja2 الوصف مثال الناتج
تنسيق الأرقام { 1234567|intcomma } { 1234567|number_format } إضافة فواصل للأرقام 1,234,567
تقريب الأرقام {{ 3.14159|floatformat:'2' }} { 3.14159|round(2) } تقريب العدد العشري 3.14
العمليات الحسابية {{ 10|add:5 }} { 10|add(5) } إجراء عملية جمع 15

فلاتر التاريخ والوقت

الفلتر DTL Jinja2 الوصف مثال الناتج
تنسيق التاريخ {{ today|date:"Y-m-d" }} { today|date('%Y-%m-%d') } تنسيق التاريخ حسب الصيغة 2023-05-15
الوقت المنقضي {{ post.date|timesince }} {{ post.date|timesince }} الوقت المنقضي منذ التاريخ 3 أيام، 4 ساعات

🛠 إنشاء Custom Filters في Django

الخطوات الأساسية

  1. إنشاء مجلد templatetags داخل تطبيقك
  2. إنشاء ملف __init__.py (فارغ)
  3. إنشاء ملف للفلاتر مثل custom_filters.py
  4. تسجيل وتنفيذ الفلاتر المخصصة

مثال 1: فلتر الضرب

# templatetags/custom_filters.py
from django import template

register = template.Library()

@register.filter(name='multiply')
def multiply(value, arg):
    """يضرب القيمة في الوسيط"""
    try:
        return float(value) * float(arg)
    except (ValueError, TypeError):
        return ''

الاستخدام في القالب:

{ load custom_filters %}

السعر بعد الضريبة: { product.price|multiply:'1.15' }

مثال 2: فلتر التمييز (Highlight)

# templatetags/custom_filters.py
from django.utils.html import escape
from django.utils.safestring import mark_safe

@register.filter(name='highlight')
def highlight(text, query):
    """يميز الكلمات المطلوبة في النص"""
    if not query:
        return text
    text = escape(text)
    query = escape(query)
    highlighted = text.replace(query, f'<span class="highlight">{query}</span>')
    return mark_safe(highlighted)

الاستخدام في القالب:

نتائج البحث: { result.text|highlight:search_query }

مثال 3: فلتر تحويل الأرقام إلى كلمات

# templatetags/custom_filters.py
@register.filter(name='number_to_words')
def number_to_words(value):
    """يحول الأرقام إلى كلمات (0-10)"""
    num_words = [
        'صفر', 'واحد', 'اثنان', 'ثلاثة', 'أربعة',
        'خمسة', 'ستة', 'سبعة', 'ثمانية', 'تسعة', 'عشرة'
    ]
    try:
        num = int(value)
        if 0 <= num <= 10:
            return num_words[num]
        return str(num)
    except (ValueError, TypeError):
        return value

الاستخدام في القالب:

لديك { notifications.count|number_to_words } إشعارات جديدة

تحذير: عند التعامل مع HTML في الفلاتر المخصصة، يجب استخدام mark_safe بحذر وتأكد من هروب (escape) كل المدخلات غير الموثوقة لمنع ثغرات XSS.

فلتر لإنشاء روابط Twitter

@register.filter(name='twitterize')
def twitterize(text):
    """يحول @username إلى روابط Twitter"""
    import re
    pattern = re.compile(r'@(\w+)')
    replaced = pattern.sub(
        r'<a href="https://twitter.com/\1" target="_blank">@\1</a>',
        escape(text)
    )
    return mark_safe(replaced)

الاستخدام:

التغريدة: { tweet.text|twitterize }

فلتر لتنسيق أرقام الهواتف

@register.filter(name='format_phone')
def format_phone(value, country='SA'):
    """يضبط تنسيق رقم الهاتف حسب البلد"""
    if not value:
        return ''
    
    # إزالة كل ما ليس رقم
    cleaned = ''.join(c for c in str(value) if c.isdigit())
    
    if country == 'SA' and len(cleaned) == 9:
        return f"+966 {cleaned[:1]} {cleaned[1:4]} {cleaned[4:7]} {cleaned[7:]}"
    elif country == 'EG' and len(cleaned) == 10:
        return f"+20 {cleaned[:2]} {cleaned[2:6]} {cleaned[6:]}"
    
    return value

الاستخدام:

الهاتف: { user.phone|format_phone:'SA' }

فلتر لتقييم تعبيرات رياضية

@register.filter(name='evaluate')
def evaluate(text):
    """يقيم تعبيرات رياضية بسيطة"""
    try:
        from ast import literal_eval
        return literal_eval(text)
    except (ValueError, SyntaxError):
        return text

الاستخدام:

ناتج المعادلة: { "3 * (4 + 5)"|evaluate }
